import java.util.HashSet;
import java.util.LinkedList;
import java.util.Scanner;

public class Main {
    private static int res = 0;
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int[] arr = new int[4];
        for (int i = 0; i < 4; i++) {
            arr[i] = sc.nextInt();
        }
        HashSet<LinkedList<Integer>> set = new HashSet<>();
        LinkedList<Integer> list = new LinkedList<>();
        boolean[] used = new boolean[4];
        dfs(arr, set, 0, list, used);

        System.out.println(res);
    }
    public static void dfs(int[] arr, HashSet set, int k, LinkedList list, boolean[] used) {
        if (k == 3) {
            if (!set.contains(list)) {
                set.add(list);
                res++;
            }
            return;
        }
        for (int i = 0; i < 4; i++) {
            if (!used[i]) {
                list.add(arr[i]);
                k++;
                used[i] = true;
                dfs(arr, set, k, list, used);
                used[i] = false;
                k--;
                list.removeLast();
            }
        }
    }
}

import java.util.LinkedList;
import java.util.Scanner;

public class Main {
    private static int res = 0;
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        String[] arr = new String[n];
        for (int i = 0; i < arr.length; i++) {
            arr[i] = sc.next();
        }
        LinkedList<String> list = new LinkedList<>();
        for (String s : arr) {
            switch (s) {
                case "NORTH":
                    if (list != null && list.contains("SOUTH")) {
                        list.remove("SOUTH");
                    }else {
                        list.add("NORTH");
                    }
                    break;
                case "SOUTH":
                    if (list != null && list.contains("NORTH")) {
                        list.remove("NORTH");
                    }else {
                        list.add("SOUTH");
                    }
                    break;
                case "EAST":
                    if (list != null && list.contains("WEST")) {
                        list.remove("WEST");
                    }else {
                        list.add("EAST");
                    }
                    break;
                case "WEST":
                    if (list != null && list.contains("EAST")) {
                        list.remove("EAST");
                    }else {
                        list.add("WEST");
                    }
            }
        }
        for (String s : list) {
            System.out.print(s + " ");
        }
    }
}
