public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int x = sc.nextInt();
        int y = sc.nextInt();
        int[][] arr = new int[x][y];
        for (int i = 0; i < x; i++) {
            for (int j = 0; j < y; j++) {
                arr[i][j] = sc.nextInt();
            }
        }
        int[][] res = new int[x][y];
        for (int i = 0; i < x; i++) {
            for (int j = 0; j < y; j++) {
                int temp = arr[i][j];
                int xSum = 0;
                int ySum = 0;
                // è®¡ç®—æ¨ªçš„å’Œ

                for (int k = 0; k < y; k++) {
                    if (k != j) {
                        xSum += arr[i][k] * 2;
                    }

            }
        }
        // æ‰¾åˆ°æœ€ä¼˜ç»“æœ
        int xMax = -1;

    public static int solution(int[][] arr){
        int xLen = arr.length, yLen = arr[0].length;
        int[] xArrSum = new int[xLen];
        int[] yArrSum = new int[yLen];
        int sum = 0;
        // è®¡ç®—xå’Œ
        for (int i = 0; i < xLen; i++) {
            int xSumTemp = 0;
            for (int j = 0; j < yLen; j++) {
                xSumTemp += arr[i][j];
            }
            xArrSum[i] = xSumTemp;
            sum += xSumTemp;
        }
        // è®¡ç®—yå’Œ
        for (int i = 0; i < yLen; i++) {
            int ySumTemp = 0;
            for (int j = 0; j < xLen; j++) {
                ySumTemp += arr[j][i];
            }
            yArrSum[i] = ySumTemp;
        }
        int res = 0;
        for (int i = 0; i < xLen; i++) {
            for (int j = 0; j < yLen; j++) {
                res = Math.max(res, sum + xArrSum[i] + yArrSum[j] - arr[i][j]);
            }
        }
        return res;
    }
