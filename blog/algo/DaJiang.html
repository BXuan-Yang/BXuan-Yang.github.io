# 0806笔试题

时间限制: 5000MS

内存限制:1048576KB

题目描述:

给定一个二维数组cropField，其中每个元素代表农田中对应位置的作物产量，每个位置上的作物经过灌溉可使产量加1倍(即对应位置作物产量数*2)，重复灌溉同一位置不能多次加倍。现在需要修建 2 条灌溉渠道，1条从上到下灌溉，1条从左到右灌溉。请确定哪行和哪列应该被落溉，使得农田中作物的总产量最大化，并输出最大化的总产量。

输入描述

int[][]cropField ={
    {2,3,1,4}
    {1,2,0,3}
    {4,2,1,7}
    {3,1,4,2}
};

输出描述

maxTotalYield=63

样例输入

4 4
2 3 1 4
1 2 0 3
4 2 1 7
3 1 4 2

样例输出

64

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int x = sc.nextInt();
        int y = sc.nextInt();
        int[][] arr = new int[x][y];
        for (int i = 0; i < x; i++) {
            for (int j = 0; j < y; j++) {
                arr[i][j] = sc.nextInt();
            }
        }
        int[][] res = new int[x][y];
        for (int i = 0; i < x; i++) {
            for (int j = 0; j < y; j++) {
                int temp = arr[i][j];
                int xSum = 0;
                int ySum = 0;
                // 计算横的和
                for (int k = 0; k < x; k++) {
                    if (k != i) {
                        ySum += arr[k][j] * 2;
                    }
                }
                // 计算列的和
                for (int k = 0; k < y; k++) {
                    if (k != j) {
                        xSum += arr[i][k] * 2;
                    }
                }
                res[i][j] = xSum + ySum + temp * 2;
            }
        }
        // 找到最优结果
        int xMax = -1;
        int yMax = -1;
        int resMax = Integer.MIN_VALUE;
        for (int i = 0; i < x; i++) {
            for (int j = 0; j < y; j++) {
                if (res[i][j] > resMax) {
                    resMax = res[i][j];
                    xMax = i;
                    yMax = j;
                }
            }
        }
        // 计算最终结果
        int sum = 0;
        for (int i = 0; i < x; i++) {
            for (int j = 0; j < y; j++) {
                if (i != xMax && j != yMax){
                    sum += arr[i][j];
                }
            }
        }
        sum += resMax;
        System.out.println(sum);
        System.out.println(solution(arr));
    }
    // 第二种解法 简单
    public static int solution(int[][] arr){
        int xLen = arr.length, yLen = arr[0].length;
        int[] xArrSum = new int[xLen];
        int[] yArrSum = new int[yLen];
        int sum = 0;
        // 计算x和
        for (int i = 0; i < xLen; i++) {
            int xSumTemp = 0;
            for (int j = 0; j < yLen; j++) {
                xSumTemp += arr[i][j];
            }
            xArrSum[i] = xSumTemp;
            sum += xSumTemp;
        }
        // 计算y和
        for (int i = 0; i < yLen; i++) {
            int ySumTemp = 0;
            for (int j = 0; j < xLen; j++) {
                ySumTemp += arr[j][i];
            }
            yArrSum[i] = ySumTemp;
        }
        int res = 0;
        for (int i = 0; i < xLen; i++) {
            for (int j = 0; j < yLen; j++) {
                res = Math.max(res, sum + xArrSum[i] + yArrSum[j] - arr[i][j]);
            }
        }
        return res;
    }

}
